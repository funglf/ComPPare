100 template <typename Func, typename... Args>
101 nvbench::benchmark_base &add_nvbench(const char *name, Func f, Args &&...args)
102 {
103 std::tuple<std::decay_t<Args>...> cargs(std::forward<Args>(args)...);
104
105 auto nvbench_wrapper = [f, cargs = std::move(cargs)](nvbench::state &st) mutable
106 {
107 comppare::plugin::nvbenchplugin::state::set_state(&st);
108 std::apply([&](auto &&...unpacked)
109 { f(std::forward<decltype(unpacked)>(unpacked)...); }, cargs);
110 };
111
112 using Callable = NvbenchCallable<decltype(nvbench_wrapper)>;
113
114 return nvbench::benchmark_manager::get()
115 .add(std::make_unique<nvbench::benchmark<Callable>>(Callable{std::move(nvbench_wrapper)}))
116 .set_name(name);
117 }
118
119 void run()
120 {
121 NVBENCH_MAIN_INITIALIZE(nvbench_argc, nvbench_argv);
122 {
123 NVBENCH_MAIN_PARSE(nvbench_argc, nvbench_argv);
124
125 NVBENCH_MAIN_PRINT_PREAMBLE(parser);
126 NVBENCH_MAIN_RUN_BENCHMARKS(parser);
127 NVBENCH_MAIN_PRINT_EPILOGUE(parser);
128
129 NVBENCH_MAIN_PRINT_RESULTS(parser);
130 } /* Tear down parser before finalization */
131 NVBENCH_MAIN_FINALIZE();
132 }
133
134 private:
135 int nvbench_argc;
136 char** nvbench_argv;
137 comppare::plugin::PluginArgParser nvbench_parser_{"--nvbench"};

181 template <class Func>
182 nvbench::benchmark_base &register_impl(const std::string &name,
183 Func &&user_fn,
184 const InTup &inputs,
185 OutTup &outs)
186 {
187 return std::apply([&](auto&&... in_vals) -> nvbench::benchmark_base& {
188 return std::apply([&](auto&&... out_vals) -> nvbench::benchmark_base& {
189 return nb_.add_nvbench(name.c_str(),
190 std::forward<Func>(user_fn),
191 std::forward<decltype(in_vals)>(in_vals)...,
192 std::forward<decltype(out_vals)>(out_vals)...);
193 }, outs);
194 }, inputs);
195 }
196
197 void initialize(int &argc, char **argv) override
198 {
199 nb_.initialize(argc, argv);
200 }
201 void run() override
202 {
203 nb_.run();
204 }
205 };
