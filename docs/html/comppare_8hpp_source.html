769 void run(int argc = 0,
770 char **argv = nullptr)
771 {
772 comppare::internal::helper::parse_args(argc, argv);
773
774 if (impls_.empty())
775 {
776 std::cerr << "\n*----------*\nNo implementations added to the ComPPare Framework.\n*----------*\n";
777 return;
778 }
779
780 outputs_.reserve(impls_.size()); // reserve space for outputs -- resize and use index works too.
781
782 print_header();
783
784 // Main loop to iterate over all implementations
785 for (size_t k = 0; k < impls_.size(); ++k)
786 {
787 // Get the current implementation
788 auto &impl = impls_[k];
789
790 OutTup outs;
791
792 double func_duration;
793 /*
794 use std::apply to unpack the inputs and outputs completely to do 1 function call of the implementation
795 this is equivalent to calling:
796 impl.fn(inputs[0], inputs[1], ..., outputs[0], outputs[1], iters, roi_us);
797 */
798 std::apply([&](auto const &...in)
799 { std::apply(
800 [&](auto &...out)
801 {
802 auto func_start = comppare::config::clock_t::now();
803 impl.fn(in..., out...);
804 auto func_end = comppare::config::clock_t::now();
805 func_duration = std::chrono::duration<double, std::micro>(func_end - func_start).count();
806 },
807 outs); },
808 inputs_);
809
810 // Calculate the time taken by the function in microseconds
811 double roi_us = comppare::config::get_roi_us();
812 double warmup_us = comppare::config::get_warmup_us();
813 double func_us = func_duration - warmup_us;
814 double ovhd_us = func_us - roi_us;
815
816 roi_us /= static_cast<double>(comppare::config::bench_iters());
817
818 PolicyTup errs{};
819 if (k)
820 {
821 compute_errors(errs, outs, *outputs_[0]);
822 }
823 outputs_.push_back(std::make_shared<OutTup>(std::move(outs)));
824 // print row
825 std::cout << comppare::internal::ansi::RESET
826 << std::left << std::setw(PRINT_COL_WIDTH) << comppare::internal::ansi::GREEN(impl.name)
827 << std::fixed << std::setprecision(2) << std::right
828 << comppare::internal::ansi::YELLOW
829 << std::setw(PRINT_COL_WIDTH) << roi_us
830 << comppare::internal::ansi::DIM
831 << std::setw(PRINT_COL_WIDTH) << func_us
832 << std::setw(PRINT_COL_WIDTH) << ovhd_us
833 << comppare::internal::ansi::RESET;
834
835 print_metrics(errs);
836 if (k && any_fail(errs))
837 std::cout << comppare::internal::ansi::BG_RED("<-- FAIL");
838 std::cout << '\n';
839
840 } /* for impls */
841
842 comppare::current_state::set_using_plugin(true);
843 if (plugin_)
844 {
845 plugin_->initialize(argc, argv);
846 plugin_->run();
847 }
848 comppare::current_state::set_using_plugin(false);
849 } /* run */
850 }; /* OutputContext */
851 }; /* InputContext */
